week = np.zeros(len(dep2train))
week2 = np.zeros(len(dep2test))

from datetime import timedelta, date

for i in range(len(dep2train)):
    if dep2train['date'].iloc[i].year==2010:
        week[i] = (dep2train['date'].dt.date.iloc[i]-date(2010,1,1)).days//7
    elif dep2train['date'].iloc[i].year==2011:
        week[i] = (dep2train['date'].dt.date.iloc[i]-date(2011,1,1)).days//7

for i in range(len(dep2test)):
    if dep2test['date'].iloc[i].year==2011:
        week2[i] = (dep2test['date'].dt.date.iloc[i]-date(2011,1,1)).days//7
    elif dep2test['date'].iloc[i].year==2012:
        week2[i] = (dep2test['date'].dt.date.iloc[i]-date(2012,1,1)).days//7

dep2train_week=dep2train
dep2train_week['week']=np.array(week)
dep2test_week=dep2test
dep2test_week['week']=np.array(week2)

def mape(y_true, y_pred): 
    return np.mean(np.abs((y_pred-y_true)/y_true))

def transf(df):
    df_new = pd.DataFrame()
    df_new['y1'] = df['weeklySales'].shift(periods=1, freq=None, axis=0)
    df_new['y2'] = df['weeklySales'].shift(periods=2, freq=None, axis=0)
    df_new['y3'] = df['weeklySales'].shift(periods=3, freq=None, axis=0)
    df_new['y4'] = df['weeklySales'].shift(periods=4, freq=None, axis=0)
    df_new['weeklySales']=df['weeklySales']
    df_new = df_new.dropna(axis=0, how='any')
    
    return df_new
