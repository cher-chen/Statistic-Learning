import numpy as np

def de_mean(ytrain, ytest):
    
    ytrain_mean = ytrain.mean()
    ytrain_dm = ytrain - ytrain_mean
    ytest_dm = ytest - ytrain_mean
    return ytrain_dm, ytest_dm

# regression function
from sklearn.linear_model import LinearRegression
from sklearn.linear_model import Lasso
from sklearn.linear_model import Ridge

def simple_regression(xtrain, ytrain, xtest, ytest):
    ols = LinearRegression().fit(xtrain, ytrain)
    ypred = ols.predict(xtest)
    rmse = np.sqrt(((ypred - ytest) ** 2).mean())
    return rmse, ypred[:20], ytest[:20].values

def ridge_regression(xtrain, ytrain, xtest, ytest, alphaList):
    best_rmse = 999
    best_alpha = -1
    ypred_best = []
    rmseList = []
    for alpha in alphaList:
        ridgereg = Ridge(alpha = alpha).fit(xtrain, ytrain) 
        ypred = ridgereg.predict(xtest)
        rmse = np.sqrt(((ypred - ytest) ** 2).mean())
        rmseList.append(rmse)
        if rmse < best_rmse:
            best_rmse = rmse
            best_alpha = alpha
            ypred_best = ypred
        
    return rmseList, best_rmse, best_alpha, ypred_best[:20], ytest[:20].values

def lasso_regression(xtrain, ytrain, xtest, ytest, alphaList):
    best_rmse = 999
    best_alpha = -1
    y_pred_best = []
    rmseList = []
    for alpha in alphaList:
        lassoreg = Lasso(alpha = alpha).fit(xtrain, ytrain) 
        ypred = lassoreg.predict(xtest)
        rmse = np.sqrt(((ypred - ytest) ** 2).mean())
        rmseList.append(rmse)
        if rmse < best_rmse:
            best_rmse = rmse
            best_alpha = alpha
            ypred_best = ypred
            
    return rmseList, best_rmse, best_alpha, ypred_best[:20], ytest[:20].values
